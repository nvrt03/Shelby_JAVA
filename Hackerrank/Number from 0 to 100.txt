Neha and Nisha are playing a game .Nisha ask Neha to enter the number N from given range of 0 to 99 .Now Nisha need to print the number till N and once it 
exceed the number from N then Print "Games End".If Neha enter the number less then 0 print Invalid input.

import java.util.Scanner;

public class Game {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int n = sc.nextInt();

    if (n < 0) {
      System.out.println("Invalid Input");
    } else {
      for (int i = 0; i <= n; i++) {
        System.out.println(i);
      }
      System.out.println("Games End");
    }
  }
}
